import random
import librosa
import pygame

# --- Audio Analysis ---
def load_audio(file_path):
    y, sr = librosa.load(file_path, sr=None)
    return y, sr

def get_beats(y, sr):
    tempo, beat_frames = librosa.beat.beat_track(y=y, sr=sr)
    beat_times = librosa.frames_to_time(beat_frames, sr=sr)
    return tempo, beat_times

# --- Level Block Generator ---
def generate_level_from_beats(beat_times, level_width, level_height):
    block_types = ['platform', 'hazard', 'pickup']
    blocks = []
    spacing = level_width / len(beat_times)
    for i, time in enumerate(beat_times):
        x = int(i * spacing)
        y = random.randint(level_height // 2, level_height - 100)
        block = {
            'x': x,
            'y': y,
            'type': random.choice(block_types),
            'color': random.choice([(255, 0, 0), (0, 255, 0), (0, 128, 255)])
        }
        blocks.append(block)
    return blocks

# --- Pygame Level Renderer ---
def render_level(screen, blocks):
    for block in blocks:
        rect = pygame.Rect(block['x'], block['y'], 60, 20)
        pygame.draw.rect(screen, block['color'], rect)

import tkinter as tk

from urllib.parse import quote

from ddgs import DDGS
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
from selenium.webdriver.common.by import By


def scrape_duckduckgo(query, max_results=10):
    results = []
    try:
        with DDGS() as ddgs:
            for r in ddgs.text(query, max_results=max_results):
                if "href" in r:
                    results.append(r["href"])
    except Exception as e:
        results.append(f"DuckDuckGo error: {e}")
    return results


def scrape_google_selenium(query, max_results=10):
    options = Options()
    options.add_argument("--headless")
    options.add_argument("--disable-gpu")
    options.add_argument("--log-level=3")
    options.add_argument("--window-size=1920,1080")

    try:
        driver = webdriver.Chrome(options=options)
        search_url = f"https://www.google.com/search?q={quote(query)}"
        driver.get(search_url)

        links = []
        elements = driver.find_elements(By.XPATH, "//div[@class='yuRUbf']/a")
        for e in elements[:max_results]:
            href = e.get_attribute("href")

                links.append(href)

        driver.quit()
        return links
    except Exception as e:
        return [f"Google Selenium Error: {e}"]


def perform_search():
    query = entry.get()
    max_results = int(result_count.get())
    output.delete(1.0, tk.END)

    output.insert(tk.END, f"Searching for: {query}\n\n")

    for engine, func in {
        "Google": scrape_google_selenium,
        "DuckDuckGo": scrape_duckduckgo
    }.items():
        output.insert(tk.END, f"--- {engine} Results ---\n")
        urls = func(query, max_results)
        for i, url in enumerate(urls, 1):
            output.insert(tk.END, f"{i}. {url}\n")
        output.insert(tk.END, "\n")
