import json
import random
import os
from datetime import datetime
from pathlib import Path

# -- Homebrew Subclasses --
homebrew_subclasses = {
    "Oath of the Fractured Sun": {
        "class": "Paladin",
        "abilities": ["Radiant Reversal", "Sunforged Step"]
    },
    "Chronothief": {
        "class": "Rogue",
        "abilities": ["Pocket Second", "Time Slash"]
    },
    "Graveborn Warden": {
        "class": "Ranger",
        "abilities": ["Soulbind Mark", "Withering Step"]
    }
}

# -- Load Monster Archive --
def load_monsters(filepath="data/monsters.json"):
    with open(filepath, "r", encoding="utf-8") as f:
        return json.load(f)

# -- Simulate Subclass Turn --
def simulate_turn(subclass_name, monster):
    subclass = homebrew_subclasses[subclass_name]
    ability = random.choice(subclass["abilities"])
    log = f"\nğŸ›¡ï¸ {subclass_name} uses {ability}!"

    if ability == "Radiant Reversal" and "radiant" in monster["type"].lower():
        log += f" Reflected radiant damage strikes {monster['name']} for 15 damage!"
        monster["hp"] -= 15

    elif ability == "Sunforged Step":
        log += " Teleports into sunlight, avoiding next hit."

    elif ability == "Pocket Second":
        log += " Delays action â€” will act twice next round."

    elif ability == "Time Slash":
        log += f" Timeline attack! {monster['name']} takes 22 damage."
        monster["hp"] -= 22

    elif ability == "Soulbind Mark":
        log += f" Necrotic tether deals 10 damage to {monster['name']}."
        monster["hp"] -= 10

    elif ability == "Withering Step":
        log += " Shrouded in mist â€” enemy speed reduced."

    return log, monster

# -- Save Encounter Log --
def save_log(encounter_log, out_dir="logs/encounters"):
    Path(out_dir).mkdir(parents=True, exist_ok=True)
    filename = datetime.now().strftime("%Y-%m-%d_%H%M%S") + ".json"
    with open(Path(out_dir) / filename, "w", encoding="utf-8") as f:
        json.dump(encounter_log, f, indent=4)
    print(f"\nğŸ“œ Log saved to {filename}")

# -- Main Encounter Runner --
def run_encounter():
    monsters = load_monsters()
    monster = random.choice(monsters)
    monster_starting_hp = monster["hp"]

    print(f"\nğŸ’€ Encounter Begins: {monster['name']} (HP: {monster['hp']}, AC: {monster['ac']})")

    log_data = {
        "monster": monster["name"],
        "start_hp": monster_starting_hp,
        "rounds": []
    }

    round_log = []

    for subclass in homebrew_subclasses:
        log, monster = simulate_turn(subclass, monster)
        print(log)
        round_log.append(log)

    log_data["end_hp"] = max(monster["hp"], 0)
    log_data["rounds"].append(round_log)

    print(f"\nğŸ§¾ Encounter Ends: {monster['name']} has {max(monster['hp'], 0)} / {monster_starting_hp} HP.")
    save_log(log_data)

# -- Entry Point --
if __name__ == "__main__":
    run_encounter()
